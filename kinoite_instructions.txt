# Kinoite

## Set up

```bash
rpm-ostree upgrade
rpm-ostree install alacritty niri xwayland-satellite ksshaskpass
```

## Install homebrew

```bash
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
echo 'eval "$(/home/linuxbrew/.linuxbrew/bin/brew shellenv)"' >> ~/.bashrc
brew install fish distrobox rofi btop podman-compose nvim pipx lazygit yazi bat task fzf chezmoi

```

## Set up dev container

```bash
distrobox create --name dev
distrobox enter dev
dnf copr enable iucar/cran
dnf copr enable iucar/rstudio
curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh
dnf install R golang julia rstudio-desktop uv R-CoprManager
cargo install --locked tree-sitter-cli
distrobox-export --bin /usr/bin/R
distrobox-export --bin /usr/bin/julia
distrobox-export --bin /usr/bin/go
distrobox-export --bin /usr/bin/uv
distrobox-export --app rstudio

exit
```

## Set up utils container

```bash
distrobox create --name utils
distrobox enter utils

dnf install zathura rofi
# Needed to launch AppImages
sudo dnf install -y fuse fuse-libs
distrobox-export --bin /usr/bin/zathura
distrobox-export --bin /usr/bin/rofi
distrobox-export --app zathura
distrobox-export --app rofi
exit
```

## Install Noctalia

```
distrobox enter utils
sudo dnf install --nogpgcheck --repofrompath 'terra,https://repos.fyralabs.com/terra$releasever' terra-release
sudo dnf install noctalia-shell
sudo tee /usr/bin/nmcli << 'EOF'
#!/bin/bash
distrobox-host-exec nmcli "$@"
EOF
sudo chmod +x /usr/bin/nmcli

sudo tee /usr/bin/niri << 'EOF'
#!/bin/bash
distrobox-host-exec niri "$@"
EOF
sudo chmod +x /usr/bin/niri

distrobox-export --bin /usr/bin/qs
exit
```

## Set up GUIs

```bash
flatpak remote-add --user --if-not-exists flathub https://flathub.org/repo/flathub.flatpakrepo
flatpak install --user flathub com.bitwarden.desktop com.spotify.Client org.zotero.Zotero org.ferdium.Ferdium app.zen_browser.zen com.calibre_ebook.calibre org.onlyoffice.desktopeditors org.mozilla.Thunderbird com.borgbase.Vorta it.mijorus.gearlever com.valvesoftware.Steam us.zoom.Zoom -y
```
