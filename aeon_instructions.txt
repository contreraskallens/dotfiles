# Set up

```bash
sudo transactional-update dup
sudo transactional-update pkg install alacritty niri xwayland-satellite
```


# Install homebrew

```bash
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
echo 'eval "$(/home/linuxbrew/.linuxbrew/bin/brew shellenv)"' >> ~/.bashrc
brew install fish distrobox rofi btop podman-compose nvim pipx lazygit yazi bat task fzf chezmoi
```

# Clone home

```bash
chezmoi init https://github.com/contreraskallens/dotfiles.git
```

```bash
sh ~/.local/share/chezmoi/scripts/init_containers_aeon.sh
```

# Set up GUIs

